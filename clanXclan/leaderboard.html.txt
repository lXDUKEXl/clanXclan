  <script>
    async function loadBoards() {
      const { data: clanWins } = await supabase.rpc('get_clan_wins');
      const { data: mvps } = await supabase.rpc('get_top_mvps');

      clanWins.forEach(c => {
        const li = document.createElement('li');
        li.textContent = `${c.name}: ${c.wins} wins`;
        document.getElementById('clanBoard').appendChild(li);
      });

      mvps.forEach(m => {
        const li = document.createElement('li');
        li.textContent = `${m.username}: ${m.mvp_count} MVPs`;
        document.getElementById('mvpBoard').appendChild(li);
      });
    }
    loadBoards();
  </script>
